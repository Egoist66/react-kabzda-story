
Component that changes its background state using reducer


export const Button: FC = () => {
  

  const {isColored, makeButtonColor} = useButtonReducer({
    isColored: false
  })

  return (

    <button style={{backgroundColor: isColored ? 'red' : 'blue'}} onClick={makeButtonColor}>
      Click me
    </button>
  )
}
